Logger

支持多线程和进程的日志输出器

快速上手 

from skylib import logger

if __name__ == '__main__':
    # 创建多进程日志输出器对象
    plogger = logger.ProcessLogger(False)
    # tlogger = logger.ThreadLogger(False)
    # ThreadLogger 用法与ProcessLogger完全相同，但是在另一线程上运行

    # 启用输出器
    plogger.startLogThread()

    # 获取包装器
    logw = plogger.getWrapperInstance('Main')
    logw.info('info')
    logw.warn('warning')
    logw.error('error')

    # 停止日志输出
    logw.stopLogger() # 输出线程/进程将会在无消息5s后退出

from skylib.logger import ProcessLogger, LoggerWrapper as _LogW
from threading import Thread
from multiprocessing import Process

def thread(_logw: _LogW) -> None:
    _logw.info('Thread.')
    pass

def process(_logw: _LogW) -> None:
    _logw.info('Process.')
    pass

if __name__ == '__main__':
    plogger = ProcessLogger(False)
    plogger.startLogThread()

    Process(target = process, args = (plogger.getWrapperInstance('Process'), )).start()
    Thread(target = thread, args = (plogger.getWrapperInstance('Thread'), )).start()

    plogger.getWrapperInstance('Main').stopLogger()

多级的输出接口 

from skylib.logger import ProcessLogger, LoggerWrapper as _LogW
from threading import Thread
from multiprocessing import Process

def thread(_logw: _LogW) -> None:
    _logw.info('Thread.')
    pass

def process(_logw: _LogW) -> None:
    _logw.info('Process.')
    pass

if __name__ == '__main__':
    plogger = ProcessLogger(False)
    plogger.startLogThread()
    logger: _LogW = plogger.getWrapperInstance('Main')

    # LoggerWrapper 对象的 getWrapperInstance 返回下一级的输出接口
    Process(target = process, args = (logger.getWrapperInstance('Process'), )).start()
    Thread(target = thread, args = (logger.getWrapperInstance('Thread'), )).start()

    logger.stopLogger()

自定义输出等级或颜色

详见logger.py文件内注释和LoggerWrapper&LoggerPacket的具体实现。

例子：

from typing import Any as _Any
from skylib.logger import LoggerPacket, ProcessLogger, LoggerWrapper as _LogW

class MyWrapper(_LogW):
    def logsPacketer(self, logs: str, sign) -> LoggerPacket:
        if sign in ['warn', 'info', 'error', 'stop']:
            return super().logsPacketer(logs, sign)
        elif sign == 'success':
            return LoggerPacket('green', self.phase_name, 'success', logs)

    def success(self, *logs: _Any, sep: str = ' ') -> None:
        log_t = sep.join([str(log) for log in logs])
        self.pushLog(self.logsPacketer(log_t, 'success'))

if __name__ == '__main__':
    plogger = ProcessLogger(False)
    plogger.changeWrapper(MyWrapper)
    plogger.startLogThread()
    logger: MyWrapper = plogger.getWrapperInstance('Main')

    logger.info('info')
    logger.warn('warn')
    logger.error('error')
    logger.success('success')

    logger.stopLogger()
